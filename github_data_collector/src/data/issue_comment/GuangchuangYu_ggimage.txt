#!/bin/env Rscript
library(ggimage)

                  type = sample(LETTERS[1:3], 10, replace=T),
                  angle = runif(10, 0, 360))

options(ggimage.keytype = "point")
ggplot(d, aes(x, y, colour=type, angle=angle)) +
    geom_pokemon(aes(image=ifelse(x>5, 'pikachu', 'tauros')))

windows()
