from ctypes import *
import sys
import rospy
import cv2
import darknet as dn

from sensor_msgs.msg import Image
from cv_bridge import CvBridge, CvBridgeError

net = dn.load_net("/home/sternl/darknet/cfg/yolov3-tiny.cfg", "/home/sternl/darknet/yolov3-tiny.weights", 0)
meta = dn.load_meta("/home/sternl/darknet/cfg/coco.data")

def callback(imgmsg):
     bridge = CvBridge()
     color_img = bridge.imgmsg_to_cv2(imgmsg, "bgr8")
     while not rospy.is_shutdown(): 
        r = dn.detect(net, meta, "/home/sternl/darknet/data/dog.jpg")
        for item in r:
            print(item)    
     cv2.imshow("realsense", color_img)
     cv2.waitKey(1)   
     rate.sleep()

if __name__ == '__main__':  
	
    rospy.init_node('listener', anonymous=True)
    rospy.Subscriber('/camera/color/image_raw', Image, callback)
    rate = rospy.Rate(10)  
    while not rospy.is_shutdown():
        rospy.spin()
